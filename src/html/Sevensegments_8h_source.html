341  uint8_t translateChar(const uint8_t glyph)
342  {
343  uint8_t i = 255; // start at -1 because the counter "pre increments" but do...while loops are faster
344 
345  do
346  {
347  i++;
348  } while (glyph != sevenSegCodes[(i << 1)]);
349 
350  return sevenSegCodes[(i << 1) +1];
351  }
