151  void displaySmallBitmap(ucoordinate_t X, ucoordinate_t Y, uint8_t H, uint8_t W, uint8_t* buffer)
152  {
153  // Tweaked by MD from the orginal by Limor "Lady Ada" Fried.
154  // 8-bits are used in preference to 16 where possible because these are faster on AtMegas
155  screen.startWrite();
156 
157  int16_t byteWidth = (W + 7) >> 3;
158  uint8_t byte {0};
159 
160  for (int8_t j {0}; j < H; j++, Y++)
161  {
162  for (int8_t i {0}; i < W ; i++)
163  {
164  if (i & 7)
165  {
166  byte <<= 1;
167  }
168  else
169  {
170  byte = pgm_read_byte(&buffer[j * byteWidth + (i >> 3)]);
171  }
172  if (! (byte & 0x80))
173  {
174  screen.writePixel(X + i, Y, m_flash);
175  }
176  }
177  }
178  screen.endWrite();
179  }
