57 inline gfxglyph_t* pgm_read_glyph_ptr(const gfxfont_t*gfxFont, uint8_t c)
58 {
59  #ifdef __AVR__
60  return &(((gfxglyph_t *)pgm_read_pointer(&gfxFont->glyph))[c]);
61  #else
62  // expression in __AVR__ section may generate "dereferencing type-punned
63  // pointer will break strict-aliasing rules" warning In fact, on other
64  // platforms (such as STM32) there is no need to do this pointer magic as
65  // program memory may be read in a usual way So expression may be simplified
66  return gfxFont->glyph + c;
67  #endif //__AVR__
68 }
